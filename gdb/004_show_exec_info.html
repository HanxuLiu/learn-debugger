显示可执行程序信息

示例代码 bubble_sort.c：

#include <stdio.h>

void BubbleSort(int *arr, int size) {
  int i, j, tmp;
  for (i = 0; i < size - 1; i++) {
    for (j = 0; j < size - i - 1; j++) {
      if (arr[j] > arr[j + 1])
      {
        tmp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = tmp;
      }
    }
  }
}

int main(int argc, char* argv[]) {
  int a[5] = { 43, 56, 78, 23, 43 };
  printf("\n排序之前输出：\n");
  for (int i = 0; i < 5; i++)
    printf("%5d", a[i]);
  printf("\n");
  BubbleSort(a, 5);
  printf("\n排序之后输出：\n");
  for (int i = 0; i < 5; i++)
    printf("%5d", a[i]);
  printf("\n\n");
  return 0;
}
